Get started in 60 seconds

# Clone and install
git clone https://github.com/mpalermiti/burnmeter.git
cd burnmeter && pip install -e .

# Add your API keys
cp .env.example .env

# Configure Claude Code (~/.claude/mcp_config.json)
{
  "mcpServers": {
    "burnmeter": {
      "command": "python",
      "args": ["src/server.py"]
    }
  }
}
